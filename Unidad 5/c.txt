TrackEvent
Endpoint: /IAnalyticsService/TrackEvent
Formato: JSON
Cabecera entrada: -
Cabecera salida: -
Método HTTP: POST
Estructura IN:
{
  "eventName": "UserSignIn",
  "properties": {
    "userId": "456",
    "device": "desktop",
    "region": "Argentina"
  }
}
Estructura OUT:
{
  "status": true
}
Estructura de datos ERR:
{
  "type": "error",
  "description": "No se pudo registrar el evento."
}



GetEventCount
Endpoint: /IAnalyticsService/GetEventCount
Formato: JSON
Cabecera entrada: -
Cabecera salida: -
Método HTTP: GET
Estructura IN:
{
  "eventName": "UserSignIn"
}
Estructura OUT:
{
  "eventName": "UserSignIn",
  "count": 275
}
Estructura de datos ERR:
{
  "type": "error",
  "description": "Evento no encontrado."
}
